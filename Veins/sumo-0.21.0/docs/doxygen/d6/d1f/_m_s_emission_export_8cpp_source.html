49 MSEmissionExport::write(OutputDevice& of, SUMOTime timestep) {
50 
51  of.openTag("timestep").writeAttr("time", time2string(timestep));
52 
53  MSVehicleControl& vc = MSNet::getInstance()->getVehicleControl();
54  MSVehicleControl::constVehIt it = vc.loadedVehBegin();
55  MSVehicleControl::constVehIt end = vc.loadedVehEnd();
56  for (; it != end; ++it) {
57  const MSVehicle* veh = static_cast<const MSVehicle*>((*it).second);
58  if (!veh->isOnRoad()) {
59  continue;
60  }
61 
62  std::string fclass = veh->getVehicleType().getID();
63  fclass = fclass.substr(0, fclass.find_first_of("@"));
64 
65  Position pos = veh->getLane()->getShape().positionAtOffset(veh->getPositionOnLane());
66  of.openTag("vehicle").writeAttr("id", veh->getID()).writeAttr("eclass", PollutantsInterface::getName(veh->getVehicleType().getEmissionClass())).writeAttr("CO2", veh->getCO2Emissions());
67  of.writeAttr("CO", veh->getCOEmissions()).writeAttr("HC", veh->getHCEmissions()).writeAttr("NOx", veh->getNOxEmissions());
68  of.writeAttr("PMx", veh->getPMxEmissions()).writeAttr("fuel", veh->getFuelConsumption()).writeAttr("noise", veh->getHarmonoise_NoiseEmissions());
69  of.writeAttr("route", veh->getRoute().getID()).writeAttr("type", fclass).writeAttr("waiting", veh->getWaitingSeconds());
70  of.writeAttr("lane", veh->getLane()->getID()).writeAttr("pos", veh->getPositionOnLane()).writeAttr("speed", veh->getSpeed());
71  of.writeAttr("angle", veh->getAngle()).writeAttr("x", pos.x()).writeAttr("y", pos.y());
72  of.closeTag();
73  }
74  of.closeTag();
75 }
