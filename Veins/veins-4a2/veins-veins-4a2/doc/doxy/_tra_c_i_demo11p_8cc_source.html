75 void TraCIDemo11p::handleParkingUpdate(cObject* obj) {
76  isParking = mobility->getParkingState();
77  if (sendWhileParking == false) {
78  if (isParking == true) {
79  (FindModule<BaseConnectionManager*>::findGlobalModule())->unregisterNic(this->getParentModule()->getSubmodule("nic"));

81  else {
82  Coord pos = mobility->getCurrentPosition();

85  }
